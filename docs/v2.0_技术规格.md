## 二、用户体系

### 2.1 登录流程

```typescript
interface AuthCredentials {
  username: string;
  password: string;
}

interface AuthSession {
  username: string;
  token: string;          // JWT token，代理服务签发
  expiresAt: number;      // token 过期时间戳
}

class AuthManager {
  private session: AuthSession | null = null;
  private proxyClient: ProxyClient;

  /**
   * 登录
   * 调用代理服务 POST /auth/login
   * 成功后将 session 缓存到本地（Office JS Settings API）
   */
  async login(credentials: AuthCredentials): Promise<boolean> {
    const response = await this.proxyClient.post('/auth/login', credentials);
    if (response.ok) {
      this.session = response.data;
      await this.saveSessionToLocal(this.session);
      return true;
    }
    return false;
  }

  /**
   * 恢复登录状态
   * Add-in 启动时调用，从本地缓存读取 session
   * 如果 token 未过期，直接恢复；否则需要重新登录
   */
  async restoreSession(): Promise<boolean> {
    const cached = await this.loadSessionFromLocal();
    if (cached && cached.expiresAt > Date.now()) {
      this.session = cached;
      return true;
    }
    return false;
  }

  /**
   * 获取当前用户名
   * 离线时仍可返回（从本地缓存读取）
   */
  getCurrentUsername(): string | null {
    return this.session?.username || null;
  }

  /**
   * 检查是否已登录
   */
  isLoggedIn(): boolean {
    return this.session !== null && this.session.expiresAt > Date.now();
  }

  /**
   * 本地存储使用 Office JS Settings API
   * 这是 Add-in 官方推荐的持久化方案，数据绑定到当前文档
   */
  private async saveSessionToLocal(session: AuthSession): Promise<void> {
    Office.context.document.settings.set('auth_session', JSON.stringify(session));
    await Office.context.document.settings.saveAsync();
  }

  private async loadSessionFromLocal(): Promise<AuthSession | null> {
    const raw = Office.context.document.settings.get('auth_session');
    return raw ? JSON.parse(raw) : null;
  }
}
```

## 三、代理服务 API 设计

### 3.1 接口总览

```
基础URL: {PROXY_BASE_URL}  （部署时配置）

认证方式: Bearer Token（JWT）
  - /auth/login 和 /auth/register 不需要 token
  - 其他接口在 Header 中携带: Authorization: Bearer {token}

通用响应格式:
  成功: { "ok": true, "data": {...} }
  失败: { "ok": false, "error": "错误描述", "code": "ERROR_CODE" }
```

```

  请求: { "username": "string", "password": "string" }
  响应: { "ok": true, "data": { "username": "string", "token": "string", "expiresAt": number } }
  错误: AUTH_INVALID_CREDENTIALS

POST /auth/register  (仅管理员可调用)
  请求: { "username": "string", "password": "string", "role": "admin" | "user" }
  响应: { "ok": true, "data": { "username": "string" } }
  错误: AUTH_USER_EXISTS

POST /auth/change-password
  请求: { "oldPassword": "string", "newPassword": "string" }
  响应: { "ok": true }
  错误: AUTH_INVALID_CREDENTIALS
```

### 3.3 AI 转发接口

```
POST /ai/chat
  请求: {
    "messages": [{ "role": "user" | "assistant", "content": "string" }],
    "feature": "table_design" | "value_analysis" | "impact_analysis" | "chat"
  }
  响应: { "ok": true, "data": { "content": "string", "usage": { "used": number, "limit": number } } }
  错误: AI_RATE_LIMITED（附带 remaining, resetAt 信息）
  
  代理服务内部逻辑:
    1. 验证 token，提取 username
    2. 检查 username + feature 的当日调用次数是否超限
    3. 补充 ANTHROPIC_API_KEY
    4. 转发到 Claude API: POST https://api.anthropic.com/v1/messages
    5. 记录调用次数
    6. 返回结果

POST /ai/analyze
  请求: {
    "tableData": {...},        // 表数据摘要（不发送完整数据，只发关键统计）
    "analysisType": "growth_curve" | "rarity_gradient" | "drop_rate" | "economy_balance" | "custom",
    "customPrompt": "string"   // analysisType 为 custom 时使用
  }
  响应: 同 /ai/chat
```

## 四、代理服务通信层

### 4.1 ProxyClient

```typescript
class ProxyClient {
  private baseUrl: string;
  private authManager: AuthManager;
  private _isConnected: boolean = false;

  /**
   * 通用请求方法
   * 自动附加 Authorization header
   * 处理网络错误和 token 过期
   */
  async request(method: string, path: string, body?: any): Promise<ProxyResponse> {
    const token = this.authManager.getToken();
    try {
      const response = await fetch(`${this.baseUrl}${path}`, {
        method,
        headers: {
          'Content-Type': 'application/json',
          ...(token ? { 'Authorization': `Bearer ${token}` } : {}),
        },
        body: body ? JSON.stringify(body) : undefined,
      });
      this._isConnected = true;
      return await response.json();
    } catch (error) {
      this._isConnected = false;
      throw new ProxyUnavailableError(error);
    }
  }

  async post(path: string, body: any): Promise<ProxyResponse> {
    return this.request('POST', path, body);
  }

  async get(path: string): Promise<ProxyResponse> {
    return this.request('GET', path);
  }

  /**
   * 检查代理服务是否可用
   * Add-in 启动时调用，之后通过心跳维持状态
   */
  async checkConnection(): Promise<boolean> {
    try {
      await this.get('/health');
      this._isConnected = true;
      return true;
    } catch {
      this._isConnected = false;
      return false;
    }
  }

  get isConnected(): boolean {
    return this._isConnected;
  }
}

interface ProxyResponse {
  ok: boolean;
  data?: any;
  error?: string;
  code?: string;
}

class ProxyUnavailableError extends Error {
  constructor(cause: any) {
    super('代理服务不可用');
    this.cause = cause;
  }
}
```

## 五、Studio 控制面板（替代「表格输出」）

### 5.1 状态管理

```typescript
interface StudioState {
  // 导出配置
  outputVersion: string;
  outputVersionNumber: number;
  versionSequence: number;
  operator: string;
  selectedTables: Set<string>;

  // 运行状态
  status: 'idle' | 'exporting' | 'uploading' | 'validating' | 'error';
  statusMessage: string;
  progress: number;

  // 结果
  lastExportResult: ExportResult | null;
  errorLogs: ErrorLog[];
}

class StudioController {
  private state: StudioState;
  private config: Config;              // v1.0 ConfigLoader 加载的配置
  private exportJob: ExportJob;        // v1.0 导出引擎
  private excelHelper: ExcelHelper;    // v1.0 工具函数

  /**
   * 初始化：从「表格输出」读取当前值填充到面板
   * 复用 v1.0 的 ConfigLoader.loadOutputSettings()
   */
  async initialize(): Promise<void> {
    const config = await configLoader.loadConfig(/* workbook */);
    this.state = {
      outputVersion: config.outputSettings.versionName,
      outputVersionNumber: config.outputSettings.versionNumber,
      versionSequence: config.outputSettings.versionSequence,
      operator: await this.getCurrentOperator(),
      selectedTables: new Set(
        Array.from(config.tablesToProcess.keys())
          .filter(k => config.tablesToProcess.get(k)!.shouldOutput)
      ),
      status: 'idle',
      statusMessage: '',
      progress: 0,
      lastExportResult: null,
      errorLogs: [],
    };
  }

  /**
   * 修改版本时同步写回「表格输出」
   * 双写：侧边栏状态 + 工作表单元格
   */
  async setOutputVersion(versionName: string): Promise<void> {
    this.state.outputVersion = versionName;

    // 写回「表格输出」的 #输出版本# 旁的单元格
    await Excel.run(async (context) => {
      const sheet = context.workbook.worksheets.getItem('表格输出');
      const marker = await excelHelper.findMarker(sheet, '#输出版本#');
      if (marker) {
        marker.getOffsetRange(0, 1).values = [[versionName]];
      }
      await context.sync();
    });

    // 联动更新版本号（从版本模板中读取）
    const template = this.config.versionTemplates.get(versionName);
    if (template) {
      await this.setOutputVersionNumber(this.state.outputVersionNumber);
    }
  }

  /**
   * 修改版本号时同步写回
   */
  async setOutputVersionNumber(versionNumber: number): Promise<void> {
    this.state.outputVersionNumber = versionNumber;

    await Excel.run(async (context) => {
      const sheet = context.workbook.worksheets.getItem('表格输出');
      const marker = await excelHelper.findMarker(sheet, '#输出版本号#');
      if (marker) {
        marker.getOffsetRange(0, 1).values = [[versionNumber]];
      }
      await context.sync();
    });
  }

  /**
   * 执行导出
   * 直接调用 v1.0 的 ExportJob.runExport()
   */
  async runExport(): Promise<ExportResult> {
    this.state.status = 'exporting';
    this.state.statusMessage = '正在导出...';

    try {
      const result = await this.exportJob.runExport();
      this.state.status = result.success ? 'idle' : 'error';
      this.state.lastExportResult = result;
      this.state.errorLogs = result.errors;
      return result;
    } catch (error) {
      this.state.status = 'error';
      throw error;
    }
  }

  /**
   * 校验并导出
   * 先运行规则校验，有错误则中断，无错误则导出
   */
  async validateAndExport(): Promise<ExportResult | ValidationResult[]> {
    // 1. 运行规则校验
    this.state.status = 'validating';
    const validationResults = await this.validationEngine.runRuleValidation(
      this.state.selectedTables
    );

    // 2. 检查是否有错误级别的结果
    const errors = validationResults.filter(r => r.severity === 'error');
    if (errors.length > 0) {
      this.state.status = 'error';
      this.state.statusMessage = `校验发现 ${errors.length} 个错误，导出已中断`;
      return validationResults; // 返回校验结果，不执行导出
    }

    // 3. 无错误，执行导出
    return await this.runExport();
  }

  /**
   * Git 推送
   * 直接调用 v1.0 的 ExportJob.runGitUpload()
   */
  async runGitUpload(): Promise<boolean> {
    this.state.status = 'uploading';
    this.state.statusMessage = '正在推送到 Git...';
    try {
      const result = await this.exportJob.runGitUpload();
      this.state.status = 'idle';
      return result;
    } catch (error) {
      this.state.status = 'error';
      throw error;
    }
  }
}
```

## 七、数据表注册管理

### 7.1 TableRegistry

```typescript
interface UnregisteredTable {
  sheetName: string;        // 工作表名
  hasConfigMarker: boolean; // 是否包含 #配置区域# 标记
}

class TableRegistry {
  /**
   * 扫描未注册的数据表
   * 遍历工作簿所有工作表，检查是否包含 #配置区域# 标记
   * 与表名对照中已注册的表比对，返回未注册的列表
   */
  async scanUnregistered(): Promise<UnregisteredTable[]> {
    const registered = await this.configLoader.loadTableList();
    const registeredNames = new Set(registered.keys());

    const unregistered: UnregisteredTable[] = [];

    await Excel.run(async (context) => {
      const sheets = context.workbook.worksheets;
      sheets.load('items/name');
      await context.sync();

      // 排除已知的元数据表和辅助表
      const excludeSheets = new Set(['表格输出', '配置设置表', '表名对照', '模板说明', '说明表']);

      for (const sheet of sheets.items) {
        if (excludeSheets.has(sheet.name) || registeredNames.has(sheet.name)) {
          continue;
        }

        // 检查是否包含 #配置区域# 标记
        const hasMarker = await this.excelHelper.findMarker(sheet, '#配置区域#') !== null;
        if (hasMarker) {
          unregistered.push({ sheetName: sheet.name, hasConfigMarker: true });
        }
      }
    });

    return unregistered;
  }

  /**
   * 注册新表到「表名对照」
```

```typescript
  private configManager: ConfigManager;

  /**
   * 创建工作表
   *

   *   Row 1: [A]version_r [B]roads_0 [C]roads_1 ... [J]roads_8 [K-L]空 [M]#配置区域# [N]字段1定义 [O]字段2定义 ...

   * 
   * 生成的布局（有 version_c）：
   *   Row 1: [K]版本列属 [L]version_c [M]startVer [N]startVer [O]startVer ...
   *   Row 2-4: 空行
   *   Row 5: version_r 行（同上）
   *   Row 6: 版本行属（同上）

  async createTable(config: TableCreationConfig): Promise<void> {
    await Excel.run(async (context) => {
      // 1. 创建新工作表
      const sheet = context.workbook.worksheets.add(config.chineseName);

      // 2. 读取线路配置（用于填充 roads 列头和中文描述）
      const lineTemplates = await this.configManager.loadAll()
        .then(c => c.lineTemplates);

      // 3. 计算列布局
      const roadsCount = lineTemplates.size;  // 通常为 9（roads_0 ~ roads_8）
      const gapCols = 2;                       // version_r/roads 和数据区之间的空列
      const configMarkerCol = 1 + roadsCount + gapCols;  // #配置区域# 所在列（从0计数）
      const dataStartCol = configMarkerCol + 1;           // 第一个数据列

      // 4. 确定 version_r 所在行
      let versionRowStart = 1;  // 默认第1行
      if (config.includeVersionCol) {
        versionRowStart = 5;  // version_c 占前几行

        // 写入 version_c 区域
        // 版本列属标记
        sheet.getCell(0, configMarkerCol - 2).values = [["版本列属"]];
        sheet.getCell(0, configMarkerCol - 1).values = [["version_c"]];
        // 每个数据列的初始版本值
        for (let i = 0; i < config.fields.length; i++) {
          sheet.getCell(0, dataStartCol + i).values = [[config.startVersion]];
        }
      }

      // 5. 写入 version_r 行
      const vrRow = versionRowStart - 1;  // 0-indexed
      sheet.getCell(vrRow, 0).values = [["version_r"]];

      // 写入 roads 列头
      const sortedLines = Array.from(lineTemplates.entries())
        .sort((a, b) => a[0] - b[0]);
      for (let i = 0; i < sortedLines.length; i++) {
        sheet.getCell(vrRow, 1 + i).values = [[sortedLines[i][1].field]];  // roads_0, roads_1, ...
      }

      // 写入 #配置区域# 标记
      sheet.getCell(vrRow, configMarkerCol).values = [["#配置区域#"]];

      // 写入字段定义
      for (let i = 0; i < config.fields.length; i++) {
        const f = config.fields[i];
        let fieldName = f.name;
        if (f.isKey) fieldName = `key_${fieldName}`;
        if (f.isLanguage) fieldName = `language_${fieldName}`;
        sheet.getCell(vrRow, dataStartCol + i).values = [[`${fieldName}=${f.type}`]];
      }

      // 6. 写入版本行属行（中文描述行）
      const descRow = vrRow + 1;
      sheet.getCell(descRow, 0).values = [["版本行属"]];

      // 写入线路中文描述
      for (let i = 0; i < sortedLines.length; i++) {
        sheet.getCell(descRow, 1 + i).values = [[sortedLines[i][1].remark]];
      }

      // 写入字段中文描述
      for (let i = 0; i < config.fields.length; i++) {
        sheet.getCell(descRow, dataStartCol + i).values = [[config.fields[i].description]];
      }

      // 7. 设置格式
      // 数据列列宽自适应
      // 冻结窗格（冻结表头行 + version_r/roads 列）
      sheet.freezePanes.freezeAt(sheet.getCell(descRow + 1, dataStartCol));

      await context.sync();

      // 8. 记录用于撤销
      this.lastCreatedSheet = config.chineseName;
    });

    // 9. 自动注册
    if (config.autoRegister) {
      await this.tableRegistry.registerTable({
        chineseName: config.chineseName,
        englishName: config.englishName,
        shouldOutput: true,
        versionRange: config.startVersion,
      });
    }
  }

  /**
   * 撤销上次创建
   * 删除工作表 + 取消注册
   */

    if (!this.lastCreatedSheet) return false;

    const sheetName = this.lastCreatedSheet;

    await Excel.run(async (context) => {
      const sheet = context.workbook.worksheets.getItem(sheetName);
      sheet.delete();
      await context.sync();
    });

    await this.tableRegistry.unregisterTable(sheetName);
    this.lastCreatedSheet = null;
    return true;
  }
```

```typescript
class ValidationEngine {
  private versionFilter: VersionFilter;  // v1.0 复用

  /**
   * 运行规则校验（离线可用）
   * 按选定的表逐个执行所有启用的规则
   */
  async runRuleValidation(
    tableNames: Set<string>,
    enabledRules?: Set<string>
  ): Promise<ValidationResult[]> {
    const results: ValidationResult[] = [];

    for (const tableName of tableNames) {
      const tableData = await this.loadTableData(tableName);
      if (!tableData) continue;

      // 格式校验
      results.push(...this.validateFormat(tableName, tableData));

      // 版本覆盖完整性
      results.push(...this.validateVersionCoverage(tableName, tableData));

      // 同Key版本顺序
      results.push(...this.validateKeyVersionOrder(tableName, tableData));

      // 必填字段
      results.push(...this.validateRequiredFields(tableName, tableData));

      // Roads 一致性
      results.push(...this.validateRoadsConsistency(tableName, tableData));
    }

    return results;
  }

  /**
   * 格式校验
   */
  private validateFormat(tableName: string, data: TableValidationData): ValidationResult[] {
    const results: ValidationResult[] = [];

    // 1. 版本区间格式检查
    for (let i = 0; i < data.versionValues.length; i++) {
      const value = String(data.versionValues[i]);
      if (value === '' || value === '版本行属') continue;

      // 检查是否使用了横线而非波浪号
      if (value.includes('-') && !value.includes('~')) {
        results.push({
          severity: 'error',
          ruleName: '版本区间分隔符',
          tableName,
          location: { sheetName: tableName, row: data.versionRowStart + i, column: 1 },
          message: `版本区间 "${value}" 使用了横线 -，应使用波浪号 ~`,
        });
        continue;
      }

      // 尝试解析版本区间
      try {
        this.versionFilter.parseRange(value);
      } catch (e) {
        results.push({
          severity: 'error',
          ruleName: '版本区间格式',
          tableName,
          location: { sheetName: tableName, row: data.versionRowStart + i, column: 1 },
          message: `版本区间 "${value}" 格式无效: ${e.message}`,
        });
      }
    }

    // 2. 数据类型检查
    for (let col = 0; col < data.fieldTypes.length; col++) {
      const expectedType = data.fieldTypes[col]; // 如 "int", "string", "int[][]"
      if (!expectedType) continue;

      for (let row = 0; row < data.dataValues.length; row++) {
        const value = data.dataValues[row][col];
        if (value === '' || value === null || value === undefined) continue;

        const typeError = this.checkDataType(String(value), expectedType);
        if (typeError) {
          results.push({
            severity: 'warning',
            ruleName: '数据类型匹配',
            tableName,
            location: {
              sheetName: tableName,
              row: data.dataRowStart + row,
              column: data.dataColStart + col
            },
            message: `字段 "${data.fieldNames[col]}" 定义为 ${expectedType}，但值 "${value}" ${typeError}`,
          });
        }
      }
    }

    return results;
  }

  /**
   * 数据类型检查辅助函数
   */
  private checkDataType(value: string, expectedType: string): string | null {
    switch (expectedType) {
      case 'int':
        if (!/^-?\d+$/.test(value)) return '不是有效整数';
        break;
      case 'float':
        if (isNaN(Number(value))) return '不是有效数字';
        break;
      case 'int[]':
        if (!/^-?\d+(\|-?\d+)*$/.test(value)) return '格式应为 N|N|N';
        break;
      case 'int[][]':
        if (!/^-?\d+(\|-?\d+)*(;-?\d+(\|-?\d+)*)*$/.test(value)) return '格式应为 N|N;N|N';
        break;
      // string 类型不校验
    }
    return null;
  }

  /**
   * 版本覆盖完整性检查（核心功能）
   * 
   * 对同一 Key 的所有行，检查版本区间是否有间隙
   */
  private validateVersionCoverage(
    tableName: string,
    data: TableValidationData
  ): ValidationResult[] {
    const results: ValidationResult[] = [];

    // 按 Key（第一数据列）分组
    const keyGroups = new Map<string, { row: number; versionRange: string }[]>();

    for (let i = 0; i < data.dataValues.length; i++) {
      const key = String(data.dataValues[i][0]); // 第一数据列
      const version = String(data.versionValues[i + 2] || ''); // +2 跳过两行表头
      if (!key || key === '') continue;

      if (!keyGroups.has(key)) keyGroups.set(key, []);
      keyGroups.get(key)!.push({
        row: data.dataRowStart + i,
        versionRange: version,
      });
    }

    // 对多行的 Key 进行覆盖检查
    for (const [key, rows] of keyGroups) {
      if (rows.length <= 1) continue; // 单行 Key 不需要检查

      // 解析每行的版本区间
      const ranges: { row: number; min: number; max: number; raw: string }[] = [];
      for (const r of rows) {
        try {
          const parsed = this.versionFilter.parseRange(r.versionRange);
          ranges.push({ row: r.row, min: parsed.min, max: parsed.max, raw: r.versionRange });
        } catch {
          // 格式错误在 validateFormat 中已报告
          continue;
        }
      }

      if (ranges.length <= 1) continue;

      // 按 min 排序
      ranges.sort((a, b2) => a.min - b2.min);

      // 检查相邻区间是否有间隙
      for (let i = 1; i < ranges.length; i++) {
        const prev = ranges[i - 1];
        const curr = ranges[i];

        if (curr.min > prev.max) {
          // 有间隙
          results.push({
            severity: 'error',
            ruleName: '版本覆盖完整性',
            tableName,
            location: { sheetName: tableName, row: curr.row, column: 1 },
            message: `Key=${key} 在版本 ${prev.max}~${curr.min} 之间无配置数据。导出该区间内的版本时此 Key 将不存在。`,
          });
        } else if (curr.min < prev.max) {
          // 有重叠（提示，非错误）
          results.push({
            severity: 'info',
            ruleName: '版本覆盖完整性',
            tableName,
            location: { sheetName: tableName, row: curr.row, column: 1 },
            message: `Key=${key} 在版本 ${curr.min}~${prev.max} 存在重叠配置，导出时将使用靠后的行（第 ${curr.row} 行）。`,
          });
        }
      }
    }

    return results;
  }

   * 必填字段检查
   */
  private validateRequiredFields(
    tableName: string,
    data: TableValidationData
  ): ValidationResult[] {
    const results: ValidationResult[] = [];
    for (let row = 0; row < data.dataValues.length; row++) {
      for (let col = 0; col < data.dataValues[row].length; col++) {
        const value = data.dataValues[row][col];
        if (value === '' || value === null || value === undefined) {
          results.push({
            severity: 'warning',
            ruleName: '必填字段',

            location: {
              sheetName: tableName,

              column: data.dataColStart + col
            },

          });
        }
      }
    }
    return results;
  }

  /**
   * Roads 一致性检查
   */
  private validateRoadsConsistency(
    tableName: string,
    data: TableValidationData
  ): ValidationResult[] {
    const results: ValidationResult[] = [];
    // 检查 roads_0 = 0 但其他 roads_X = 1 的矛盾情况
    for (let i = 0; i < data.roadsValues.length; i++) {
      const roads = data.roadsValues[i]; // [roads_0, roads_1, ..., roads_8]
      if (!roads || roads.length === 0) continue;

      const roads0 = String(roads[0]);
      if (roads0 === '0' || roads0 === '') {
        // roads_0 为 0，检查其他线路是否有为 1 的
        for (let j = 1; j < roads.length; j++) {
          if (String(roads[j]) === '1') {
            results.push({
              severity: 'warning',
              ruleName: 'Roads一致性',
              tableName,
              location: { sheetName: tableName, row: data.versionRowStart + i + 2, column: 2 },
              message: `第 ${data.versionRowStart + i + 2} 行 roads_0=0（总线路禁用），但 roads_${j} =1，该行在所有版本中都不会导出`,
            });
            break;
          }
        }
      }
    }
    return results;
  }
```

```typescript
  async highlightInExcel(previewResult: PreviewResult): Promise<void> {
    await Excel.run(async (context) => {
      const sheet = context.workbook.worksheets.getItem(previewResult.tableName);

      // 清除旧的高亮
      await this.clearHighlights(sheet, context);

      // 绿色：保留的行
      // （不主动设置，保持原样即可，排除的行变灰就够了）

      // 灰色：排除的行
      for (const rowIdx of previewResult.excludedRows) {
        const row = sheet.getRangeByIndexes(rowIdx - 1, 0, 1, 30); // 覆盖足够列数
        row.format.fill.color = '#E0E0E0';
        row.format.font.color = '#999999';
      }

      // 黄色：被覆盖的重复Key行
      for (const rowIdx of previewResult.overriddenRows) {
        const row = sheet.getRangeByIndexes(rowIdx - 1, 0, 1, 30);
        row.format.fill.color = '#FFF3CD';
        row.format.font.strikethrough = true;
      }

      // 灰色：排除的列（如果有 version_c）
      for (const colIdx of previewResult.excludedCols) {
        const col = sheet.getRangeByIndexes(0, colIdx - 1, 200, 1); // 覆盖足够行数
        col.format.fill.color = '#E0E0E0';
      }

      await context.sync();
    });
  }

  /**
   * 清除所有高亮
   */
  async clearAllHighlights(sheetName: string): Promise<void> {
    await Excel.run(async (context) => {
      const sheet = context.workbook.worksheets.getItem(sheetName);
      const usedRange = sheet.getUsedRange();
      usedRange.format.fill.clear();
      usedRange.format.font.color = '#000000';
      usedRange.format.font.strikethrough = false;
      await context.sync();
    });
  }
```

### 11.2 远程指令接收端（本地桌面端）

```typescript
class RemoteCommandListener {
  private proxyClient: ProxyClient;
  private authManager: AuthManager;
  private studioController: StudioController;
  private pollInterval: number = 5000;    // 轮询间隔 5 秒
  private heartbeatInterval: number = 30000; // 心跳间隔 30 秒
  private pollTimer: any = null;
  private heartbeatTimer: any = null;
  private requireConfirmation: boolean = false; // 可配置

  /**
   * 启动监听
   * Add-in 启动且登录成功后调用
   */
  start(): void {
    // 启动指令轮询
    this.pollTimer = setInterval(() => this.pollCommands(), this.pollInterval);

    // 启动心跳
    this.heartbeatTimer = setInterval(() => this.sendHeartbeat(), this.heartbeatInterval);
    this.sendHeartbeat(); // 立即发送第一次
  }

  /**
   * 停止监听
   */
  stop(): void {
    if (this.pollTimer) clearInterval(this.pollTimer);
    if (this.heartbeatTimer) clearInterval(this.heartbeatTimer);
  }

  /**
   * 轮询待执行的指令
   */
  private async pollCommands(): Promise<void> {
    if (!this.proxyClient.isConnected) return;

    try {
      const response = await this.proxyClient.get('/commands/poll');
      if (!response.ok || !response.data.commands.length) return;

      for (const cmd of response.data.commands) {
        await this.handleCommand(cmd);
      }
    } catch {
      // 网络错误，静默忽略，下次轮询重试
    }
  }

  /**
   * 处理单个指令
   */
  private async handleCommand(cmd: RemoteCommand): Promise<void> {
    // 检查是否需要确认
    if (this.requireConfirmation) {
      const confirmed = await this.showConfirmation(cmd);
      if (!confirmed) {
        // 用户拒绝，回报取消状态
        await this.reportResult(cmd.commandId, 'failed', '本地用户拒绝执行');
        return;
      }
    }

    // 回报开始执行
    await this.reportResult(cmd.commandId, 'started', '正在执行...');

    try {
      switch (cmd.command) {
        case 'export':
          // 先设置导出参数
          if (cmd.params.outputVersion) {
            await this.studioController.setOutputVersion(cmd.params.outputVersion);
          }
          if (cmd.params.outputVersionNumber) {
            await this.studioController.setOutputVersionNumber(cmd.params.outputVersionNumber);
          }
          // 执行导出
          const exportResult = await this.studioController.runExport();
          await this.reportResult(cmd.commandId, 'completed', '导出完成', exportResult);
          break;

        case 'git_upload':
          const gitResult = await this.studioController.runGitUpload();
          await this.reportResult(cmd.commandId, gitResult ? 'completed' : 'failed',
            gitResult ? 'Git 推送完成' : 'Git 推送失败');
          break;
      }
    } catch (error) {
      await this.reportResult(cmd.commandId, 'failed', `执行失败: ${error.message}`);
    }
  }

  /**
   * 回报执行结果
   */
  private async reportResult(
    commandId: string,
    status: string,
    message: string,
    result?: any
  ): Promise<void> {
    try {
      await this.proxyClient.post('/commands/result', { commandId, status, message, result });
    } catch { /* 网络错误，静默 */ }
  }

  /**
   * 发送心跳
   */
  private async sendHeartbeat(): Promise<void> {
    try {
      await this.proxyClient.post('/heartbeat', { clientType: 'desktop' });
    } catch { /* 静默 */ }
  }

  /**
   * 显示确认提示
   * 使用 Office JS Dialog API 或侧边栏内的确认弹窗
   */
  private async showConfirmation(cmd: RemoteCommand): Promise<boolean> {
    // 在侧边栏显示确认提示
    // 返回用户是否确认
    return new Promise((resolve) => {
      // 触发 React 组件的确认弹窗
      this.onConfirmationRequired?.(cmd, resolve);
    });
  }

  /**
   * 确认回调（由 UI 层设置）
   */
  onConfirmationRequired?: (cmd: RemoteCommand, callback: (confirmed: boolean) => void) => void;
}

interface RemoteCommand {
  commandId: string;
  fromUsername: string;
  command: 'export' | 'git_upload';
  params: {
    outputVersion?: string;
    outputVersionNumber?: number;
  };
  createdAt: number;
}
```

### 11.3 OneDrive 回退模式

```typescript
class OneDriveFallbackListener {
  private authManager: AuthManager;
  private studioController: StudioController;
  private changeHandler: any = null;

  /**
   * 启动 OneDrive 回退监听
   * 当代理服务不可用时激活
   * 
   * 监听「表格输出」工作表中 #操作标识# 旁的单元格变化
   */
  async start(): Promise<void> {
    await Excel.run(async (context) => {
      const sheet = context.workbook.worksheets.getItem('表格输出');

      // 注册 onChanged 事件
      this.changeHandler = sheet.onChanged.add(async (event) => {
        await this.handleSheetChange(event);
      });

      await context.sync();
    });
  }

  /**
   * 停止监听
   */
  async stop(): Promise<void> {
    if (this.changeHandler) {
      await Excel.run(async (context) => {
        this.changeHandler.remove();
        await context.sync();
      });
    }
  }

  /**
   * 处理工作表变化事件
   * 逻辑与 VBA 的 Worksheet_Calculate 对应
   */
  private async handleSheetChange(event: Excel.WorksheetChangedEventArgs): Promise<void> {
    await Excel.run(async (context) => {
      const sheet = context.workbook.worksheets.getItem('表格输出');

      // 读取操作标识
      const operationMarker = await excelHelper.findMarker(sheet, '#操作标识#');
      if (!operationMarker) return;

      const operationCell = operationMarker.getOffsetRange(0, 1);
      operationCell.load('values');

      const statusMarker = await excelHelper.findMarker(sheet, '#工作状态#');
      if (!statusMarker) return;

      const statusCell = statusMarker.getOffsetRange(0, 1);
      statusCell.load('values');

      await context.sync();

      const operation = String(operationCell.values[0][0]);
      const status = String(statusCell.values[0][0]);

      // 解析操作：格式为 "操作|用户名"
      const pipePos = operation.indexOf('|');
      if (pipePos === -1) return;

      const operationText = operation.substring(0, pipePos);
      const operatorName = operation.substring(pipePos + 1);

      if (operationText === '空' || operationText === '') return;
      if (status !== '空' && status !== '') return;

      // v2.0 改进：用登录用户名匹配，替代机器码
      const currentUser = this.authManager.getCurrentUsername();
      if (currentUser !== operatorName) return;

      // 匹配成功，执行操作
      statusCell.values = [['运行中']];
      await context.sync();

      try {
        switch (operationText) {
          case '导出':
            await this.studioController.runExport();
            break;
          case '上传':
            await this.studioController.runGitUpload();
            break;
          case '重置':
            // 重置 UI 状态
            break;
        }
      } finally {
        // 清空操作标识和状态
        statusCell.values = [['空']];
        operationCell.values = [['空']];
        await context.sync();
      }
    });
  }
}
```

### 11.4 协作模式自动切换

```typescript
class CollaborationManager {
  private proxyClient: ProxyClient;
  private commandListener: RemoteCommandListener;
  private fallbackListener: OneDriveFallbackListener;
  private currentMode: 'proxy' | 'onedrive' | 'offline' = 'offline';

  /**
   * 初始化：检测代理服务可用性，选择模式
   */
  async initialize(): Promise<void> {
    const proxyAvailable = await this.proxyClient.checkConnection();

    if (proxyAvailable) {
      this.switchToProxyMode();
    } else {
      this.switchToOneDriveMode();
    }

    // 持续检测代理服务状态
    setInterval(() => this.checkAndSwitch(), 30000); // 每30秒检查一次
  }

  private async checkAndSwitch(): Promise<void> {
    const proxyAvailable = await this.proxyClient.checkConnection();

    if (proxyAvailable && this.currentMode !== 'proxy') {
      this.switchToProxyMode();
    } else if (!proxyAvailable && this.currentMode === 'proxy') {
      this.switchToOneDriveMode();
    }
  }

  private switchToProxyMode(): void {
    this.fallbackListener.stop();
    this.commandListener.start();
    this.currentMode = 'proxy';
    // 通知 UI 更新状态栏
  }

  private switchToOneDriveMode(): void {
    this.commandListener.stop();
    this.fallbackListener.start();
    this.currentMode = 'onedrive';
    // 通知 UI 更新状态栏
  }

  get mode(): string {
    return this.currentMode;
  }
}
```
