### 2.2 StudioController

```typescript
class StudioController {
  private state: StudioState;
  private config: Config;
  private exportJob: ExportJob;  // v1.0 导出引擎

  /**
   * 初始化：从「表格输出」读取当前值填充到面板
   * 实现版本号记忆——上次使用的值自动成为默认值
   */
  async initialize(): Promise<void> {
    const config = await configLoader.loadConfig();
    this.state = {
      outputVersion: config.outputSettings.versionName,
      outputVersionNumber: config.outputSettings.versionNumber,
      versionSequence: config.outputSettings.versionSequence,
      operator: await this.getCurrentOperator(),
      selectedTables: new Set(
        Array.from(config.tablesToProcess.keys())
          .filter(k => config.tablesToProcess.get(k)!.shouldOutput)
      ),
      status: 'idle',
      statusMessage: '',
      progress: 0,
      lastExportResult: null,
      errorLogs: [],
    };
  }

  /**
   * 修改版本时同步写回「表格输出」
   * 双写：侧边栏状态 + 工作表单元格（向后兼容 VBA）
   */
  async setOutputVersion(versionName: string): Promise<void> {
    this.state.outputVersion = versionName;
    await Excel.run(async (context) => {
      const sheet = context.workbook.worksheets.getItem('表格输出');
      const marker = await excelHelper.findMarker(sheet, '#输出版本#');
      if (marker) {
        marker.getOffsetRange(0, 1).values = [[versionName]];
      }
      await context.sync();
    });
  }

  /**
   * 修改版本号时同步写回
   */
  async setOutputVersionNumber(versionNumber: number): Promise<void> {
    this.state.outputVersionNumber = versionNumber;
    await Excel.run(async (context) => {
      const sheet = context.workbook.worksheets.getItem('表格输出');
      const marker = await excelHelper.findMarker(sheet, '#输出版本号#');
      if (marker) {
        marker.getOffsetRange(0, 1).values = [[versionNumber]];
      }
      await context.sync();
    });
  }

  /**
   * 执行导出（直接调用 v1.0 ExportJob）
   */
  async runExport(): Promise<ExportResult> {
    this.state.status = 'exporting';
    this.state.statusMessage = '正在导出...';
    try {
      const result = await this.exportJob.runExport();
      this.state.status = result.success ? 'idle' : 'error';
      this.state.lastExportResult = result;
      this.state.errorLogs = result.errors;
      return result;
    } catch (error) {
      this.state.status = 'error';
      throw error;
    }
  }

  /**
   * Git 推送（直接调用 v1.0 ExportJob）
   */
  async runGitUpload(): Promise<boolean> {
    this.state.status = 'uploading';
    try {
      const result = await this.exportJob.runGitUpload();
      this.state.status = 'idle';
      return result;
    } catch (error) {
      this.state.status = 'error';
      throw error;
    }
  }
}
```

### 3.1 ConfigManager

```typescript
class ConfigManager {
  private configLoader: ConfigLoader;  // v1.0 加载器
  private excelHelper: ExcelHelper;

  /** 读取配置（复用 v1.0） */
  async loadAll(): Promise<Config> {
    return await this.configLoader.loadConfig();
  }

  /**
   * 添加版本模板
   * 在「配置设置表」#版本列表# 区域最后一行下方追加
   */
  async addVersion(version: VersionTemplate): Promise<void> {
    await Excel.run(async (context) => {
      const sheet = context.workbook.worksheets.getItem('配置设置表');
      const marker = await this.excelHelper.findMarker(sheet, '#版本列表#');
      if (!marker) throw new Error('未找到 #版本列表# 标记');
      const region = marker.getResizedRange(0, 0).getSurroundingRegion();
      region.load('rowCount');
      await context.sync();
      const newRow = marker.getOffsetRange(region.rowCount, 0);
      newRow.getResizedRange(0, 2).values = [[version.name, version.lineId, version.gitDirectory]];
      await context.sync();
    });
  }

  /** 修改版本模板 */
  async updateVersion(oldName: string, updated: VersionTemplate): Promise<void>;

  /** 删除版本模板（清空对应行） */
  async deleteVersion(name: string): Promise<void>;

  /** 线路模板的增删改（操作 #线路列表# 区域） */
  async addLine(line: LineTemplate): Promise<void>;
  async updateLine(id: number, updated: LineTemplate): Promise<void>;

  /** 人员代码的增删改（操作 #人员代码# 区域） */
  async addStaff(staff: StaffInfo): Promise<void>;
  async updateStaff(name: string, updated: StaffInfo): Promise<void>;

  /** Git 提交模板（操作 #Git通用提交日志# 下一行） */
  async setGitCommitTemplate(template: string): Promise<void>;

  /** 功能开关（操作 #配置开关# 区域） */
  async setSwitch(name: string, value: boolean): Promise<void>;
}

interface StaffInfo {
```

## 四、数据表注册管理

### 4.1 TableRegistry

```typescript
interface UnregisteredTable {
  sheetName: string;
  hasConfigMarker: boolean;  // 是否包含 #配置区域#
}

class TableRegistry {
  /**
   * 扫描未注册的数据表
   * 遍历所有工作表，检查是否含 #配置区域# 但未在表名对照中注册
   */
  async scanUnregistered(): Promise<UnregisteredTable[]> {
    const registered = await this.configLoader.loadTableList();
    const registeredNames = new Set(registered.keys());
    const unregistered: UnregisteredTable[] = [];
    const excludeSheets = new Set(['表格输出', '配置设置表', '表名对照', '模板说明', '说明表']);

    await Excel.run(async (context) => {
      const sheets = context.workbook.worksheets;
      sheets.load('items/name');
      await context.sync();

      for (const sheet of sheets.items) {
        if (excludeSheets.has(sheet.name) || registeredNames.has(sheet.name)) continue;
        const hasMarker = await this.excelHelper.findMarker(sheet, '#配置区域#') !== null;
        if (hasMarker) {
          unregistered.push({ sheetName: sheet.name, hasConfigMarker: true });
        }
      }
    });
    return unregistered;
  }

  /**
   * 注册新表到「表名对照」
   */
  async registerTable(info: TableInfo): Promise<void> {
    await Excel.run(async (context) => {
      const sheet = context.workbook.worksheets.getItem('表名对照');
      const marker = await this.excelHelper.findMarker(sheet, '#输出控制#');
      if (!marker) throw new Error('未找到 #输出控制# 标记');
      const region = marker.getResizedRange(0, 0).getSurroundingRegion();
      region.load('rowCount');
      await context.sync();
      const newRow = marker.getOffsetRange(region.rowCount, 0);
      newRow.getResizedRange(0, 3).values = [
        [info.versionRange, info.chineseName, info.englishName, info.shouldOutput]
      ];
      await context.sync();
    });
  }

  /** 修改已注册表 */
  async updateTable(chineseName: string, updates: Partial<TableInfo>): Promise<void>;

  /** 取消注册（清空行，不删除工作表） */
  async unregisterTable(chineseName: string): Promise<void>;
}
```

## 五、新表创建向导

### 5.1 TableCreator

```typescript
interface TableCreationConfig {
  chineseName: string;
  englishName: string;
  startVersion: string;
  fields: FieldDefinition[];
  includeVersionCol: boolean;  // 是否包含 version_c
  autoRegister: boolean;       // 是否自动注册到表名对照
}

interface FieldDefinition {
  name: string;           // 如 "id"
  type: string;           // 如 "int"
  description: string;    // 如 "怪物ID"
  isKey: boolean;         // 加 key_ 前缀
  isLanguage: boolean;    // 加 language_ 前缀
}

class TableCreator {
  private lastCreatedSheet: string | null = null;  // 用于撤销

  /**
   * 创建工作表
   *
   * 无 version_c 时生成：
   *   Row 1: version_r | roads_0~roads_8 | 空列 | #配置区域# | 字段定义...
   *   Row 2: 版本行属  | 线路中文名...  | 空列 |            | 中文描述...
   *
   * 有 version_c 时生成：
   *   Row 1: 空 | 版本列属 | version_c | 各列版本值...
   *   Row 2-4: 空行
   *   Row 5: version_r 行（同上）
   *   Row 6: 版本行属行（同上）
   */
  async createTable(config: TableCreationConfig): Promise<void> {
    await Excel.run(async (context) => {
      const sheet = context.workbook.worksheets.add(config.chineseName);

      // 读取线路配置
      const allConfig = await this.configManager.loadAll();
      const lineTemplates = allConfig.lineTemplates;

      // 计算列布局
      const roadsCount = lineTemplates.size;  // 通常 9
      const gapCols = 2;
      const configMarkerCol = 1 + roadsCount + gapCols;
      const dataStartCol = configMarkerCol + 1;

      // 确定 version_r 所在行
      let vrRow = 0;  // 0-indexed
      if (config.includeVersionCol) {
        vrRow = 4;
        // 写入 version_c 区域
        sheet.getCell(0, configMarkerCol - 2).values = [["版本列属"]];
        sheet.getCell(0, configMarkerCol - 1).values = [["version_c"]];
        for (let i = 0; i < config.fields.length; i++) {
          sheet.getCell(0, dataStartCol + i).values = [[config.startVersion]];
        }
      }

      // 写入 version_r + roads 列头
      sheet.getCell(vrRow, 0).values = [["version_r"]];
      const sortedLines = Array.from(lineTemplates.entries()).sort((a, b) => a[0] - b[0]);
      for (let i = 0; i < sortedLines.length; i++) {
        sheet.getCell(vrRow, 1 + i).values = [[sortedLines[i][1].field]];
      }

      // #配置区域# + 字段定义
      sheet.getCell(vrRow, configMarkerCol).values = [["#配置区域#"]];
      for (let i = 0; i < config.fields.length; i++) {
        const f = config.fields[i];
        let fieldName = f.name;
        if (f.isKey) fieldName = `key_${fieldName}`;
        if (f.isLanguage) fieldName = `language_${fieldName}`;
        sheet.getCell(vrRow, dataStartCol + i).values = [[`${fieldName}=${f.type}`]];
      }

      // 版本行属 + 中文描述
      const descRow = vrRow + 1;
      sheet.getCell(descRow, 0).values = [["版本行属"]];
      for (let i = 0; i < sortedLines.length; i++) {
        sheet.getCell(descRow, 1 + i).values = [[sortedLines[i][1].remark]];
      }
      for (let i = 0; i < config.fields.length; i++) {
        sheet.getCell(descRow, dataStartCol + i).values = [[config.fields[i].description]];
      }

      // 冻结窗格
      sheet.freezePanes.freezeAt(sheet.getCell(descRow + 1, dataStartCol));
      await context.sync();

      this.lastCreatedSheet = config.chineseName;
    });

    // 自动注册
    if (config.autoRegister) {
      await this.tableRegistry.registerTable({
        chineseName: config.chineseName,
        englishName: config.englishName,
        shouldOutput: true,
        versionRange: config.startVersion,
      });
    }
  }

  /**
   * 撤销上次创建（删除工作表 + 取消注册）
   */
  async undoLastCreation(): Promise<boolean> {
    if (!this.lastCreatedSheet) return false;
    const name = this.lastCreatedSheet;
    await Excel.run(async (context) => {
      context.workbook.worksheets.getItem(name).delete();
      await context.sync();
    });
    await this.tableRegistry.unregisterTable(name);
    this.lastCreatedSheet = null;
    return true;
  }
}
```

### 6.1 监听机制

替代 VBA 的 `Worksheet_Calculate`，使用 Office JS 的 `onChanged` 事件：

```typescript
class OneDriveCollaborationListener {
  private changeHandler: any = null;

  /**
   * 启动监听
   * 监听「表格输出」工作表中 #操作标识# 旁的单元格变化
   */
  async start(): Promise<void> {
    await Excel.run(async (context) => {
      const sheet = context.workbook.worksheets.getItem('表格输出');
      this.changeHandler = sheet.onChanged.add(async (event) => {
        await this.handleChange(event);
      });
      await context.sync();
    });
  }

  async stop(): Promise<void> {
    if (this.changeHandler) {
      await Excel.run(async (context) => {
        this.changeHandler.remove();
        await context.sync();
      });
    }
  }

  /**
   * 处理变化事件
   * 逻辑对应 VBA 的 Worksheet_Calculate + IsCurrentComputer
   */
  private async handleChange(event: Excel.WorksheetChangedEventArgs): Promise<void> {
    await Excel.run(async (context) => {
      const sheet = context.workbook.worksheets.getItem('表格输出');

      // 读取操作标识
      const opMarker = await excelHelper.findMarker(sheet, '#操作标识#');
      if (!opMarker) return;
      const opCell = opMarker.getOffsetRange(0, 1);
      opCell.load('values');

      // 读取工作状态
      const statusMarker = await excelHelper.findMarker(sheet, '#工作状态#');
      if (!statusMarker) return;
      const statusCell = statusMarker.getOffsetRange(0, 1);
      statusCell.load('values');

      await context.sync();

      const operation = String(opCell.values[0][0]);
      const status = String(statusCell.values[0][0]);

      // 解析 "操作|操作人" 格式
      const pipePos = operation.indexOf('|');
      if (pipePos === -1) return;
      const operationText = operation.substring(0, pipePos);
      const operatorName = operation.substring(pipePos + 1);

      if (operationText === '空' || operationText === '') return;
      if (status !== '空' && status !== '') return;

      // 身份匹配：沿用机器码方式
      // 读取配置设置表中的人员代码，比对当前机器的主板序列号
      const isCurrentMachine = await this.checkMachineId(operatorName);
      if (!isCurrentMachine) return;

      // 匹配成功，执行操作
      statusCell.values = [['运行中']];
      await context.sync();

      try {
        switch (operationText) {
          case '导出':
            await this.studioController.runExport();
            break;
          case '上传':
            await this.studioController.runGitUpload();
            break;
          case '重置':
            // 重置 UI
            break;
        }
      } finally {
        statusCell.values = [['空']];
        opCell.values = [['空']];
        await context.sync();
      }
    });
  }

  /**
   * 机器码匹配
   * 对应 VBA 的 IsCurrentComputer 函数
   * 注意：Office JS 环境获取硬件信息受限
   * 可能需要通过 Node.js 后端辅助获取，或使用 Windows 用户名作为替代
   */

    // 实现方式待定，取决于 Office JS 运行时的能力
    // 方案A：通过 Add-in 的 Node.js 后端执行 WMI 查询（桌面版）
```
