减少配置错误（特别是版本区间错误导致的数据遗漏），提升导出信心。

### 2.2 ValidationEngine

```typescript
class ValidationEngine {
  private versionFilter: VersionFilter;  // v1.0

  /**
   * 运行全部规则校验
   * 仅手动触发（点击「校验」按钮）
   */
  async runValidation(tableNames: Set<string>): Promise<ValidationResult[]> {
    const results: ValidationResult[] = [];

    for (const tableName of tableNames) {
      const data = await this.loadTableData(tableName);
      if (!data) continue;

      results.push(...this.validateVersionFormat(tableName, data));
      results.push(...this.validateDataTypes(tableName, data));
      results.push(...this.validateArrayFormats(tableName, data));
      results.push(...this.validateVersionCoverage(tableName, data));
      results.push(...this.validateKeyVersionOrder(tableName, data));
      results.push(...this.validateRequiredFields(tableName, data));
      results.push(...this.validateRoadsConsistency(tableName, data));
    }

    return results;
  }

  // ──────────── 格式校验 ────────────

  /**
   * 版本区间格式检查
   */
  private validateVersionFormat(tableName: string, data: TableValidationData): ValidationResult[] {
    const results: ValidationResult[] = [];

    for (let i = 0; i < data.versionValues.length; i++) {
      const value = String(data.versionValues[i]);
      if (value === '' || value === '版本行属') continue;

      // 横线检查
      if (value.includes('-') && !value.includes('~')) {
        results.push({
          severity: 'error', ruleName: '版本区间分隔符', tableName,
          location: { sheetName: tableName, row: data.versionRowStart + i, column: 1 },
          message: `版本区间 "${value}" 使用了横线 -，应使用波浪号 ~`,
        });
        continue;
      }

      // 解析校验
      try {
        this.versionFilter.parseRange(value);
      } catch (e) {
        results.push({
          severity: 'error', ruleName: '版本区间格式', tableName,
          location: { sheetName: tableName, row: data.versionRowStart + i, column: 1 },
          message: `版本区间 "${value}" 格式无效: ${e.message}`,
        });
      }
    }
    return results;
  }

  /**
   * 数据类型检查
   */
  private validateDataTypes(tableName: string, data: TableValidationData): ValidationResult[] {
    const results: ValidationResult[] = [];

    for (let col = 0; col < data.fieldTypes.length; col++) {
      const expectedType = data.fieldTypes[col];
      if (!expectedType) continue;

      for (let row = 0; row < data.dataValues.length; row++) {
        const value = data.dataValues[row][col];
        if (value === '' || value === null || value === undefined) continue;

        const error = this.checkType(String(value), expectedType);
        if (error) {
          results.push({
            severity: 'warning', ruleName: '数据类型', tableName,
            location: { sheetName: tableName, row: data.dataRowStart + row, column: data.dataColStart + col },
            message: `字段 "${data.fieldNames[col]}" 定义为 ${expectedType}，但值 "${value}" ${error}`,
          });
        }
      }
    }
    return results;
  }

  private checkType(value: string, type: string): string | null {
    switch (type) {
      case 'int': return /^-?\d+$/.test(value) ? null : '不是有效整数';
      case 'float': return isNaN(Number(value)) ? '不是有效数字' : null;
      case 'int[]': return /^-?\d+(\|-?\d+)*$/.test(value) ? null : '格式应为 N|N|N';
      case 'int[][]': return /^-?\d+(\|-?\d+)*(;-?\d+(\|-?\d+)*)*$/.test(value) ? null : '格式应为 N|N;N|N';
    }
    return null; // string 等不校验
  }

  /**
   * 数组分隔符检查
   */
  private validateArrayFormats(tableName: string, data: TableValidationData): ValidationResult[] {
    const results: ValidationResult[] = [];

    for (let col = 0; col < data.fieldTypes.length; col++) {
      const type = data.fieldTypes[col];
      if (!type || !type.includes('[]')) continue;

      for (let row = 0; row < data.dataValues.length; row++) {
        const value = String(data.dataValues[row][col] || '');
        if (value === '') continue;

        // 检查是否使用了逗号而非 | 和 ;
        if (value.includes(',') && !value.includes('|')) {
          results.push({
            severity: 'warning', ruleName: '数组分隔符', tableName,
            location: { sheetName: tableName, row: data.dataRowStart + row, column: data.dataColStart + col },
            message: `字段 "${data.fieldNames[col]}" 使用了逗号分隔，应使用 | 和 ; 分隔`,
          });
        }
      }
    }
    return results;
  }

  // ──────────── 逻辑校验 ────────────

  /**
   * 版本覆盖完整性检查（核心功能）
   *
   * 对同一 Key 的所有行，检查版本区间是否连续无间隙。
   */
  private validateVersionCoverage(tableName: string, data: TableValidationData): ValidationResult[] {
    const results: ValidationResult[] = [];

    // 按 Key（第一数据列）分组
    const keyGroups = new Map<string, { row: number; versionRange: string }[]>();

    for (let i = 0; i < data.dataValues.length; i++) {
      const key = String(data.dataValues[i][0]);
      const version = String(data.versionValues[i + 2] || ''); // +2 跳过两行表头
      if (!key) continue;
      if (!keyGroups.has(key)) keyGroups.set(key, []);
      keyGroups.get(key)!.push({ row: data.dataRowStart + i, versionRange: version });
    }

    // 对多行 Key 检查覆盖
    for (const [key, rows] of keyGroups) {
      if (rows.length <= 1) continue;

      const ranges: { row: number; min: number; max: number }[] = [];
      for (const r of rows) {
        try {
          const parsed = this.versionFilter.parseRange(r.versionRange);
          ranges.push({ row: r.row, min: parsed.min, max: parsed.max });
        } catch { continue; }
      }
      if (ranges.length <= 1) continue;

      ranges.sort((a, b2) => a.min - b2.min);

      for (let i = 1; i < ranges.length; i++) {
        const prev = ranges[i - 1];
        const curr = ranges[i];

        if (curr.min > prev.max) {
          // 有间隙
          results.push({
            severity: 'error', ruleName: '版本覆盖完整性', tableName,
            location: { sheetName: tableName, row: curr.row, column: 1 },
            message: `Key=${key} 在版本 ${prev.max}~${curr.min} 之间无配置数据，导出该区间版本时此 Key 将不存在`,
          });
        } else if (curr.min < prev.max) {
          // 有重叠
          results.push({
            severity: 'info', ruleName: '版本覆盖完整性', tableName,
            location: { sheetName: tableName, row: curr.row, column: 1 },
            message: `Key=${key} 在版本 ${curr.min}~${prev.max} 存在重叠，导出时使用靠后的行（第 ${curr.row} 行）`,
          });
        }
      }
    }
    return results;
  }

  /**
   * 同Key版本顺序检查
   */
  private validateKeyVersionOrder(tableName: string, data: TableValidationData): ValidationResult[] {
    const results: ValidationResult[] = [];
    const keyGroups = new Map<string, { row: number; version: number }[]>();

    for (let i = 0; i < data.dataValues.length; i++) {
      const key = String(data.dataValues[i][0]);
      const verStr = String(data.versionValues[i + 2] || '');
      if (!key) continue;
      try {
        const parsed = this.versionFilter.parseRange(verStr);
        if (!keyGroups.has(key)) keyGroups.set(key, []);
        keyGroups.get(key)!.push({ row: data.dataRowStart + i, version: parsed.min });
      } catch { continue; }
    }

    for (const [key, rows] of keyGroups) {
      if (rows.length <= 1) continue;
      for (let i = 1; i < rows.length; i++) {
        if (rows[i].version < rows[i - 1].version) {
          results.push({
            severity: 'warning', ruleName: '同Key版本顺序', tableName,
            location: { sheetName: tableName, row: rows[i].row, column: 1 },
            message: `Key=${key} 第 ${rows[i].row} 行的版本号比第 ${rows[i-1].row} 行小，简写模式下可能导致覆盖逻辑异常`,
          });
        }
      }
    }
    return results;
  }

  /**
   * 必填字段检查
   */
  private validateRequiredFields(tableName: string, data: TableValidationData): ValidationResult[] {
    const results: ValidationResult[] = [];
    for (let row = 0; row < data.dataValues.length; row++) {
      for (let col = 0; col < data.dataValues[row].length; col++) {
        const value = data.dataValues[row][col];
        if (value === '' || value === null || value === undefined) {
          results.push({
            severity: 'warning', ruleName: '必填字段', tableName,
            location: { sheetName: tableName, row: data.dataRowStart + row, column: data.dataColStart + col },
            message: `第 ${data.dataRowStart + row} 行 "${data.fieldNames[col]}" 字段为空`,
          });
        }
      }
    }
    return results;
  }

  /**
   * Roads 一致性检查
   */
  private validateRoadsConsistency(tableName: string, data: TableValidationData): ValidationResult[] {
    const results: ValidationResult[] = [];
    for (let i = 0; i < data.roadsValues.length; i++) {
      const roads = data.roadsValues[i];
      if (!roads || roads.length === 0) continue;
      const roads0 = String(roads[0]);
      if (roads0 === '0' || roads0 === '') {
        for (let j = 1; j < roads.length; j++) {
          if (String(roads[j]) === '1') {
            results.push({
              severity: 'warning', ruleName: 'Roads一致性', tableName,
              location: { sheetName: tableName, row: data.versionRowStart + i + 2, column: 2 },
              message: `第 ${data.versionRowStart + i + 2} 行 roads_0=0（总线路禁用），但 roads_${j}=1，该行在所有版本中都不会导出`,
            });
            break;
          }
        }
      }
    }
    return results;
  }

  /**
   * 加载单表的校验数据
   */

    // 复用 v1.0 的 DataLoader 读取表数据
    // 解析字段定义行获取 fieldNames 和 fieldTypes
    // 返回结构化的校验数据
  }
}

interface TableValidationData {
  versionRowStart: number;      // version_r 所在行号（1-indexed）
  dataRowStart: number;         // 第一行数据行号（version_r + 2）
  dataColStart: number;         // 第一数据列列号（#配置区域# + 1）
  versionValues: string[];      // A 列版本区间值
  roadsValues: string[][];      // B-J 列 roads 值

  fieldTypes: string[];         // 字段类型
  dataValues: any[][];          // 数据区域值（不含表头）
}
```

```typescript
class ValidationNavigator {
  /** 跳转到指定单元格并临时高亮 */
  async navigateTo(location: CellLocation): Promise<void> {
    await Excel.run(async (context) => {
      const sheet = context.workbook.worksheets.getItem(location.sheetName);
      sheet.activate();
      const cell = sheet.getCell(location.row - 1, location.column - 1);
      cell.select();
      cell.format.fill.color = '#FFD700';
      await context.sync();

      // 3秒后清除高亮
      setTimeout(async () => {
        await Excel.run(async (ctx) => {
          ctx.workbook.worksheets.getItem(location.sheetName)
            .getCell(location.row - 1, location.column - 1)
            .format.fill.clear();
          await ctx.sync();
        });
      }, 3000);
    });
  }
}
```

### 3.1 VersionPreviewer

```typescript
interface PreviewResult {
  tableName: string;
  originalRows: number;
  originalCols: number;
  filteredRows: number;
  filteredCols: number;
  excludedRows: number[];       // 被排除的行号（1-indexed）
  excludedCols: number[];       // 被排除的列号（1-indexed）
  overriddenRows: number[];     // 被重复Key覆盖的行号
}

class VersionPreviewer {
  private dataFilter: DataFilter;      // v1.0
  private versionFilter: VersionFilter; // v1.0

  /**
   * 预览指定版本的筛选结果（不执行导出）
   */
  async preview(
    versionName: string,
    versionNumber: number,
    tableNames: Set<string>
  ): Promise<PreviewResult[]> {
    const results: PreviewResult[] = [];

    for (const tableName of tableNames) {
      const tableData = await dataLoader.loadTableData(tableName);
      if (!tableData) continue;

      const filterResult = this.dataFilter.applyFilters(
        tableData.mainData, tableData.versionRowData, tableData.versionColData,
        { targetVersion: versionNumber, lineField: this.getLineField(versionName) }
      );

      results.push({
        tableName,
        originalRows: tableData.mainData.length,
        originalCols: tableData.mainData[0]?.length || 0,
        filteredRows: filterResult.rowCount,
        filteredCols: filterResult.colCount,
        excludedRows: filterResult.excludedRowIndices,
        excludedCols: filterResult.excludedColIndices,
        overriddenRows: filterResult.overriddenRowIndices,
      });
    }
    return results;
  }

  /**
   * 在 Excel 中高亮显示
   */
  async highlightInExcel(result: PreviewResult): Promise<void> {
    await Excel.run(async (context) => {
      const sheet = context.workbook.worksheets.getItem(result.tableName);

      // 灰色：排除的行
      for (const rowIdx of result.excludedRows) {
        const row = sheet.getRangeByIndexes(rowIdx - 1, 0, 1, 30);
        row.format.fill.color = '#E0E0E0';
        row.format.font.color = '#999999';
      }

      // 黄色+删除线：被覆盖的重复Key行
      for (const rowIdx of result.overriddenRows) {
        const row = sheet.getRangeByIndexes(rowIdx - 1, 0, 1, 30);
        row.format.fill.color = '#FFF3CD';
        row.format.font.strikethrough = true;
      }

      // 灰色：排除的列
      for (const colIdx of result.excludedCols) {
        const col = sheet.getRangeByIndexes(0, colIdx - 1, 200, 1);
        col.format.fill.color = '#E0E0E0';
      }

      await context.sync();
    });
  }

  /**
   * 清除所有高亮
   */
  async clearHighlights(sheetName: string): Promise<void> {
    await Excel.run(async (context) => {
      const sheet = context.workbook.worksheets.getItem(sheetName);
      const usedRange = sheet.getUsedRange();
      usedRange.format.fill.clear();
      usedRange.format.font.color = '#000000';
      usedRange.format.font.strikethrough = false;
      await context.sync();
    });
  }
}
```

### 4.1 整合到 StudioController

```typescript
// 在 v2.0 的 StudioController 中新增方法

class StudioController {
  // ... v2.0 已有方法 ...

  private validationEngine: ValidationEngine;  // v3.0 新增

  /**
   * 一键全流程：校验 → 导出 → Git 推送
   * 任何一步失败自动中断
   */
  async runFullPipeline(): Promise<PipelineResult> {
    const result: PipelineResult = { steps: [], success: false };

    // 步骤1：校验
    this.state.status = 'validating';
    this.state.statusMessage = '正在校验...';
    const validationResults = await this.validationEngine.runValidation(this.state.selectedTables);
    const errors = validationResults.filter(r => r.severity === 'error');

    result.steps.push({ name: '校验', success: errors.length === 0, detail: `${validationResults.length} 条结果，${errors.length} 个错误` });

    if (errors.length > 0) {
      this.state.status = 'error';
      this.state.statusMessage = `校验发现 ${errors.length} 个错误，流程中断`;
      result.validationResults = validationResults;
      return result;
    }

    // 步骤2：导出
    this.state.statusMessage = '正在导出...';
    try {
      const exportResult = await this.runExport();
      result.steps.push({ name: '导出', success: exportResult.success, detail: `${exportResult.modifiedFiles.length} 个文件变更` });
      if (!exportResult.success) {
        result.exportResult = exportResult;
        return result;
      }
      result.exportResult = exportResult;
    } catch (error) {
      result.steps.push({ name: '导出', success: false, detail: error.message });
      return result;
    }

    // 步骤3：Git 推送
    this.state.statusMessage = '正在推送到 Git...';
    try {
      const gitSuccess = await this.runGitUpload();
      result.steps.push({ name: 'Git推送', success: gitSuccess, detail: gitSuccess ? '推送成功' : '推送失败' });
    } catch (error) {
      result.steps.push({ name: 'Git推送', success: false, detail: error.message });
      return result;
    }

    result.success = true;
    this.state.status = 'idle';
    this.state.statusMessage = '一键发布完成';
    return result;
  }
}

interface PipelineResult {
  steps: { name: string; success: boolean; detail: string }[];
  success: boolean;
  validationResults?: ValidationResult[];
  exportResult?: ExportResult;
}
```

## 八、测试重点

### 版本覆盖完整性检查的测试用例

```typescript
// 正常：单行
{ key: "100000", rows: [{ version: "1" }] }  // → 无告警

// 正常：简写模式连续
{ key: "3", rows: [{ version: "1" }, { version: "1.3" }] }  // → 无告警

// 错误：有间隙
{ key: "3", rows: [{ version: "1~1.2" }, { version: "1.5" }] }
// → error: Key=3 在 1.2~1.5 无数据

// 提示：有重叠
{ key: "3", rows: [{ version: "1~2.0" }, { version: "1.5" }] }
// → info: Key=3 在 1.5~2.0 重叠

// 正常：空值不导出
```
